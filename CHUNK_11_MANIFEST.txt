================================================================================
API EXAMPLES
================================================================================

GET ALL ACTIVITIES (default 20)
  curl "http://localhost:3001/api/activities"

  Response:
  {
    "activities": [
      {
        "id": "clp...",
        "type": "hire",
        "description": "Alice Johnson hired as Frontend Engineer",
        "employeeId": "emp1",
        "deviceId": null,
        "createdAt": "2024-03-20T10:00:00Z",
        "employee": {
          "id": "emp1",
          "name": "Alice Johnson",
          "email": "alice@example.com"
        },
        "device": null
      }
    ],
    "limit": 20,
    "offset": 0,
    "count": 20
  }

GET ACTIVITIES BY TYPE
  curl "http://localhost:3001/api/activities?type=hire"

GET ACTIVITIES BY EMPLOYEE
  curl "http://localhost:3001/api/activities?employeeId=emp1"

GET WITH PAGINATION
  curl "http://localhost:3001/api/activities?limit=50&offset=20"

CREATE ACTIVITY
  curl -X POST "http://localhost:3001/api/activities" \
    -H "Content-Type: application/json" \
    -d '{
      "type": "hire",
      "description": "Alice Johnson hired",
      "employeeId": "emp1"
    }'
